# 数组

## 删除排序数组中的重复项

```js
function removeDuplicates(nums) {
  if(!nums || nums.length === 0) {
    reutrn 0
  }

  let num = 0

  for (let i = 1; i < nums.length; i++) {
    if(nums[num] !== nums[i]) {
      // 下面两行可以简写为 nums[++num] = nums[i]
      nums[num + 1] = nums[i]
      num += 1
    }
  }

  return num + 1
}
```
